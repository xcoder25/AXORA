rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isSelf(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function callerProfile() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function hasCallerProfile() {
      return isSignedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function callerSchoolId() {
      return callerProfile().data.schoolId;
    }

    function isSchoolStaff() {
      return hasCallerProfile()
        && callerProfile().data.role in ['admin', 'teacher'];
    }

    function isSameSchoolAsCaller() {
      return hasCallerProfile()
        && resource.data.schoolId == callerSchoolId();
    }

    match /schools/{schoolId} {
      allow get: if true;
      allow list: if isSignedIn();
      allow create: if isSignedIn()
        && request.resource.data.adminUid == request.auth.uid;
      allow update: if isSignedIn()
        && resource.data.adminUid == request.auth.uid;
      allow delete: if false;
    }

    match /users/{userId} {
      allow get: if isSelf(userId)
        || (isSchoolStaff() && isSameSchoolAsCaller());
      allow list: if isSchoolStaff() && isSameSchoolAsCaller();
      allow create: if isSelf(userId)
        && request.resource.data.uid == userId;
      allow update: if isSelf(userId)
        || (isSchoolStaff() && isSameSchoolAsCaller());
      allow delete: if false;
    }

    match /attendance/{recordId} {
      allow read, write: if isSignedIn();
    }

    match /finance/{recordId} {
      allow read, write: if isSignedIn();
    }

    match /exams/{examId} {
      allow read, write: if isSignedIn();
    }

    match /workflows/{workflowId} {
      allow read, write: if isSignedIn();
    }

    match /announcements/{annId} {
      allow read: if isSignedIn()
        && resource.data.schoolId == callerSchoolId();
      allow create: if isSchoolStaff()
        && request.resource.data.schoolId == callerSchoolId();
      allow update, delete: if isSchoolStaff()
        && resource.data.schoolId == callerSchoolId();
    }

    match /finance_transactions/{txId} {
      allow create: if isSignedIn()
        && request.resource.data.schoolId == callerSchoolId()
        && request.resource.data.studentId == request.auth.uid;
      allow read: if isSignedIn()
        && (resource.data.studentId == request.auth.uid
          || (isSchoolStaff() && resource.data.schoolId == callerSchoolId()));
      allow update, delete: if false;
    }
  }
}
